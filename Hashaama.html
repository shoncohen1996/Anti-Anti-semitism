function generateMailtoLink() {

    var email = document.getElementById('email').value;

    var subject = document.getElementById('subject').value;

    var body = document.getElementById('body').value;


    var mailtoLink = document.getElementById('mailtoLink');

    mailtoLink.href = `mailto:${encodeURIComponent(email)}?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;

    mailtoLink.innerHTML = 'Click here to send your email!';

    mailtoLink.style.display = 'block';

}
